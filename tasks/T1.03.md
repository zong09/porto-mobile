## Spec
```dart
import 'package:drift/drift.dart';
import 'tables.dart';
part 'database.g.dart';

@DriftDatabase(tables: [Portfolios, Assets, Transactions, Liabilities,
  LiabilityTransactions, NetWorthHistory, PriceCache, Settings])
class AppDatabase extends _$AppDatabase {
  AppDatabase(super.e);
  @override
  int get schemaVersion => 1;
  @override
  MigrationStrategy get migration => MigrationStrategy(
    onCreate: (m) => m.createAll(),
    beforeOpen: (details) async { await customStatement('PRAGMA foreign_keys = ON'); },
  );
}
```
Do NOT add DAOs here (later tasks add `part` DAO files or separate DAO classes).
