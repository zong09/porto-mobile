```dart
class PricePoint {
  final int t;        // epoch ms
  final double p;     // price
  const PricePoint(this.t, this.p);
}

class PriceHistoryClient {
  final Dio dio;
  final Future<double> Function() getFx;
  PriceHistoryClient(this.dio, {required this.getFx});

  Future<List<PricePoint>> cryptoHistory(String coinId, int days);   // THB, Binance klines
  Future<List<PricePoint>> stockHistory(String symbol, String range);// range '7D'|'1M'|'3M'|'1Y'
}
```

## Dio usage rules
- Use `Options(validateStatus: (_) => true)`; branch on `resp.statusCode == 200`, else throw `ApiError`.
- `resp.data` is decoded JSON (`List` for klines, `Map` for the Yahoo chart).
- klines close is a String at index 4 → `double.parse(k[4] as String)`. Timestamp `k[0]` is an int.

## Constants
- `SYMBOL_REGEX = RegExp(r'^[A-Za-z0-9.\-^=]{1,15}$')` (validate coinId/symbol before URL).

## Algorithm (port EXACTLY)
```
cryptoHistory(coinId, days):
  validate coinId; invalid → throw ApiError
  if coinId == 'USDT':
    fx = await getFx()
    now = DateTime.now().millisecondsSinceEpoch
    return [ PricePoint(now - 86400000*days, fx), PricePoint(now, fx) ]
  # interval/limit mapping
  if days <= 1:  interval='5m';  limit=288
  elif days <= 7: interval='1h'; limit=168
  elif days <= 90: interval='1d'; limit=days
  else: interval='1w'; limit=(days/7).ceil()
  url = 'https://api.binance.com/api/v3/klines?symbol=' + Uri.encodeComponent('${coinId}USDT')
        + '&interval=$interval&limit=$limit'
  resp = dio.get(url, validateStatus all)
  if resp.statusCode != 200: throw ApiError('crypto history $coinId')
  fx = await getFx()
  return [ for k in resp.data:  PricePoint((k[0] as num).toInt(), double.parse(k[4] as String) * fx) ]

stockHistory(symbol, range):
  validate symbol; invalid → throw ApiError
  interval = range == '1Y' ? '1wk' : '1d'
  mappedRange = {'7D':'7d','1M':'1mo','3M':'3mo','1Y':'1y'}[range] ?? '3mo'
  url = 'https://query1.finance.yahoo.com/v8/finance/chart/' + Uri.encodeComponent(symbol)
        + '?range=$mappedRange&interval=$interval'
  resp = dio.get(url, options: Options(headers:{'User-Agent': UA}, validateStatus:(_)=>true))
  if resp.statusCode != 200: throw ApiError('stock history $symbol')
  res = resp.data['chart']?['result']?[0]
  ts  = res?['timestamp'] as List?          # seconds
  cl  = res?['indicators']?['quote']?[0]?['close'] as List?
  out = <PricePoint>[]
  if ts != null and cl != null:
    for i in 0..ts.length-1:
      if cl[i] != null: out.add(PricePoint((ts[i] as num).toInt()*1000, (cl[i] as num).toDouble()))
  return out
```
`UA = 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/120.0.0.0 Safari/537.36'`

## Test vectors (`test/price_history_client_test.dart`) — `class MockDio extends Mock implements Dio {}`
`setUpAll(() => registerFallbackValue(Options()));`

1. **cryptoHistory**: `getFx` returns 30. klines call → 200, data =
   `[[1000,'0','0','0','2350000','0'], [2000,'0','0','0','2360000','0']]`.
   `cryptoHistory('BTC', 30)` → 2 points: `[0].t==1000`, `[0].p` closeTo(2350000*30, 0.01); `[1].t==2000`, `[1].p` closeTo(2360000*30, 0.01).
2. **cryptoHistory USDT**: `getFx` returns 30. `cryptoHistory('USDT', 7)` → `length==2`, both `.p == 30.0` (do not assert timestamps).
3. **stockHistory skips null closes**: chart call → 200, data =
   `{'chart':{'result':[{'timestamp':[1,2,3],'indicators':{'quote':[{'close':[10.0,null,12.0]}]}}]}}`.
   `stockHistory('AAPL','1M')` → 2 points: `[(1000,10.0),(3000,12.0)]` (t = seconds*1000, null dropped).
4. **non-200 throws**: klines → statusCode 500 → `cryptoHistory('BTC',30)` throws `ApiError`.
