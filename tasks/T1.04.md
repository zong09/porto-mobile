## Spec — plain wrapper class (NOT @DriftAccessor)
```dart
import 'package:drift/drift.dart';
import 'database.dart';

class PortfolioAssetDao {
  final AppDatabase db;
  PortfolioAssetDao(this.db);

  // portfolios
  Future<void> upsertPortfolio(PortfoliosCompanion c) => db.into(db.portfolios).insertOnConflictUpdate(c);
  Future<List<Portfolio>> allPortfolios() =>
      (db.select(db.portfolios)..orderBy([(t) => OrderingTerm(expression: t.sortOrder)])).get();
  Stream<List<Portfolio>> watchPortfolios() =>
      (db.select(db.portfolios)..orderBy([(t) => OrderingTerm(expression: t.sortOrder)])).watch();
  Future<void> deletePortfolio(String id) =>
      (db.delete(db.portfolios)..where((t) => t.id.equals(id))).go();
  Future<void> reorderPortfolios(List<String> idsInOrder); // set sortOrder = index for each id

  // assets
  Future<void> upsertAsset(AssetsCompanion c) => db.into(db.assets).insertOnConflictUpdate(c);
  Future<List<Asset>> assetsFor(String portfolioId) =>
      (db.select(db.assets)..where((t) => t.portfolioId.equals(portfolioId))
        ..orderBy([(t) => OrderingTerm(expression: t.sortOrder)])).get();
  Stream<List<Asset>> watchAssetsFor(String portfolioId) => /* same as above, .watch() */;
  Future<void> deleteAsset(String id) =>
      (db.delete(db.assets)..where((t) => t.id.equals(id))).go();
}
```
`Portfolio`/`Asset` here are Drift's generated row classes (from database.dart), NOT the freezed models.
`reorderPortfolios`: loop, for each id at index i run an update setting sortOrder=i (use a batch).
